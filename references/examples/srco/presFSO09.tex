\section{Introducción}


\subsection*{Concepto}


\begin{frame}{Concepto}
	\begin{itemize}
		\item
			Comunicaciones ópticas a través del espacio libre.
		\item
			Transmisión de pulsos visibles o en el \vs{ir} modulados electrónicamente a través del aire o del espacio libre.
		\item
			Transmisión de la información en el dominio óptico, por tanto, presenta muy altas tasas binarias.
		\item
			Mismo principio de funcionamiento que los mandos a distancia o los teclados inalámbricos por \vs{ir}.
	\end{itemize}
\end{frame}


\subsection*{Funcionamiento}


\begin{frame}{Funcionamiento}
	\begin{itemize}
		\item
			Transmisión de pulsos ópticos en el IR.
		\item
			Uso de láseres de bajo consumo como emisores y detectores fotónicos de alta sensibilidad como receptores.
		\item
			Modulación electrónica del láser, enviando información heterogénea (vídeo, voz, datos, archivos\ldots)
		\item
			Empleo de lentes y telescopios para focalizar al máximo el haz de luz y conseguir el mejor alineamiento posible emisor-receptor.
	\end{itemize}
\end{frame}

\subsection*{Ventajas de \vs{fso} frente a \vs{rf}}


\begin{frame}{Ventajas de \vs{fso} frente a \vs{rf}}
	\begin{itemize}
		\item
			Mayores tasas binarias
		\item
			Equipos más baratos y sencillos que los sistemas RF de alta frecuencia.
		\item
			Menor consumo de potencia.
		\item
			Sistema muy seguro y casi imposible de <<pinchar>>.
	\end{itemize}
\end{frame}

\section{Efectos limitantes}


\begin{frame}{Ecuación de pérdidas}
	\begin{itemize}
		\item
			La transmisión por la atmósfera de las señales ópticas implica tener en cuenta las condiciones extremas que presenta el aire como medio. Varios son los fenómenos que limitan los enlaces de FSO a unos pocos kilómetros. Además existen unos factores geométricos a considerar a la hora de realizar el balance de potencias. La ecuación que sigue resume lo aquí expuesto:
	\end{itemize}
	\begin{block}{Pérdidas}
		\begin{equation}
			P_R = P_T\,\frac{A_{RX}}{(\theta l)^2}\,\text{e}^{-\alpha L}
		\end{equation}
	\end{block}
\end{frame}


\subsection*{Atenuación atmosférica}


\begin{frame}{Atenuación atmosférica}
	\begin{itemize}
		\item
			Procesos de absorción, difracción y dispersión producidos por la interactuación con las moléculas del aire y partículas suspendidas en el mismo.
		\item
			Como resultado se produce una severa atenuación que se puede disparar si se producen condiciones adversas como lluvia o niebla.
		\item
			Los fenómenos atmósfericos a tener en cuenta a la hora de planificar un enlace FSO son los siguientes:
	\end{itemize}
\end{frame}


\subsubsection*{Lluvia}


\begin{frame}{Lluvia}
	\begin{itemize}
		\item
			Produce altas atenuaciones, mayores cuanto mayor es la intensidad de lluvia. La figura que sigue muestra resultados experimentales:

\subsubsection*{Niebla}


\begin{frame}{Niebla}
	\begin{itemize}
		\item

		\item
			Existen varios modelos que estiman la atenuación producida por la niebla. Destacan el modelo de Kim, el de Kruse y el de Nabulsi.

\subsubsection*{Escintilación}


\begin{frame}{Escintilación}
	\begin{itemize}
		\item
			Este fenómeno presenta menos problemas para los enlaces FSO.
		\item
			Se produce en días soleados y calurosos, que, como resultado de la radiación solar, se forman zonas de distinto índice de refracción en la atmósfera.
		\item
			Puede llegar a generar atenuaciones de hasta 12 dB.
	\end{itemize}
\end{frame}


\subsection*{Atenuación geométrica}


\begin{frame}{Atenuación geométrica}
	\begin{itemize}
		\item
			Producido por la propia naturaleza y geometría de los emisores y receptores telescópicos empleados.
		\item
			La siguiente ecuación estima la atenuación que se genera. Dependiente de los diámetros de los telescopios y del ángulo de divergencia:
	\end{itemize}
	\begin{block}{Pérdidas}
		\begin{equation}
			Att_{geo} = \left(\frac{d_{RX}}{d_{TX} + \theta l}\right)^2
		\end{equation}
	\end{block}
\end{frame}

\subsection*{Técnicas de compensación}


\begin{frame}{Técnicas para combatir estos efectos}
	\begin{itemize}
		\item
			Existen varios mecanismos para mitigar los problemas producidos por los fenómenos atmosféricos.
		\item
			Para paliar las grandes atenuaciones producidas por la lluvia y la niebla, que pueden llegar a cortar la comunicación óptica, se suele utilizar la misma técnica.
		\item
			Esta técnica consiste en agregar un enlace <<back up>> de RF, con una tasa binaria que  permita suplir la comunicación óptica. Este enlace presentará una atenuación mucho menor.
	\end{itemize}
\end{frame}

\section{Sistemas comerciales}


\subsection*{Características generales}


\begin{frame}{Esquema de transmisión opaco}
	\begin{itemize}
		\item<1->
		Recepción de la señal en el equipo.
		\item<2->
		Conversión al dominio electrónico si es necesario.
		\item<3->
		Modulación de corriente de un láser \vs{vcsel}.
		\item<4->
		Detección directa mediante un fotodiodo de avalancha \vs{apd}.
	\end{itemize}
\end{frame}

\begin{frame}{Cumplimiento de la normativa}
	\begin{itemize}
		\item
			Los equipos deben encontrarse en la categoría 1\vs{m} definida en el estándar \vs{iec/en} 60825-1/\vs{A}2.
		\item
			Para cumplir los requisitos impuestos por el estándar:
		\begin{itemize}
			\item<1->
				Trabajar en longitudes de onda alejadas del espectro visible.
			\item<2->
				Reducir la potencia de emisión.
		\end{itemize}
	\end{itemize}
\end{frame}

\subsection*{Características adicionales}


\begin{frame}{Técnicas de compensación de fallos}
	\begin{itemize}
		\item
			Sistemas de calefacción que impiden que las ventanas de la carcasa se empañen por vaho o nieve.\pause
		\item
			Sistemas de apuntado automático guiados por sistemas de seguimiento de haz.
		\item
			Anchos de haz moderados. Solución de compromiso entre potencia y pérdida de la comunicación por falta de alineamiento.
		\item
			Transceptores múltiples para aprovechar las bondades de la diversidad multicamino.
	\end{itemize}
\end{frame}

\begin{frame}{Protección frente al entorno}
	\begin{itemize}
		\item<1->
			Carcasa de aluminio sellada herméticamente, protege al equipo frente a la deposición de partículas externas, y de las precipitaciones
		\item<2->
			Sistema de acondicionamiento térmico interno, mantiene el láser a una temperatura de 35 ºC.
		\item<3->
			Filtros solares que impiden la actuación del sol sobre los detectores.
	\end{itemize}
\end{frame}



\section{Aplicaciones}


\subsection*{Sistemas comerciales}


\begin{frame}{Uso generalizado}
	\begin{itemize}[<+->]
		\item
			Instalación rápida y sencilla. Gran ancho de banda.
		\item
			Permiten desde enlaces punto a punto hasta redes malladas.
		\item
			Rápida recuperación de la inversión.
		\item
			Especialmente explotado en el ámbito militar, empresarial y por instituciones públicas y privadas.
	\end{itemize}
\end{frame}

\begin{frame}{Despliegues rápidos}
	\begin{itemize}
		\item

		\begin{itemize}[<+->]
			\item
				Despliegue rápido en zonas catastróficas
			\item
				Despliegue móvil, para la retransmisiones de señal audiovisual de alta definición.
			\item
				Restablecimiento de enlaces críticos.
		\end{itemize}
	\end{itemize}
\end{frame}

\subsection*{City-Wide Wireless Networks}


\begin{frame}{Necesidades de ancho de banda}
	\begin{itemize}
		\item
			Aumento de la demanda de servicios de banda ancha.\par
		\item
			Limitaciones de las comunicaciones de radiofrecuencia:
		\begin{itemize}
			\item<1->
				El espectro radioeléctrico en el que se dan estas comunicaciones se ha llevado el límite.
			\item<2->
				Altos niveles de congestión e interferencias.
			\item<3->
				Necesidad de licencias en ciertas bandas de operación.
			\item<4->
				Encarecimiento de los equipos (ya de por sí costosos).
			\item<5->
				Necesidad de \vs{los} para establecer enlaces de banda ancha.
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Necesidades prácticas}
	\begin{itemize}
		\item<1->
			La topología actual comprende pequeñas redes aisladas confinadas a las estaciones base.
		\item<2->
			Topología inadecuada para dar cobertura inalámbrica a toda una ciudad.
		\item<3->
			Gran interés en sustituir la topología actual por una configuración de multicast virtual.
		\item<4->
			Para ello es necesario dotar al sistema de una red de backbone con capacidad similar a las redes de fibra.
	\end{itemize}
\end{frame}

\subsection*{Intersatellite Link}

\begin{frame}{Enlaces intersatelitales basados en \vs{fso}}
	\begin{itemize}
		\item
			Las comunicaciones intersatelitales aprovechan al máximo las ventajas de las comunicaciones \vs{fso}.
		\item
			Permiten satisfacer la demanda global de intercambio de información a altas velocidades.
		\item
			Esto se debe a que en el espacio desaparecen los efectos limitantes relacionados con la atmósfera terrestre.
		\item
			Existen dos tipos de enlaces:
		\begin{itemize}
			\item
				Entre estaciones situadas en órbitas \vs{geo}.
			\item
				Aquellos en los que al menos una de las estaciones se encuentra en una órbita distinta a la geoestacionaria.
		\end{itemize}
	\end{itemize}
\end{frame}

\subsection*{WDMFSO}

\begin{frame}{Wavelength Division Multiplexing systems on Free Space Optics communications}
	\begin{itemize}[<+->]
		\item
			La modulación óptico-electro-óptica introduce un cuello de botella en las comunicaciones \vs{fso}.
		\item
			Solución experimental factible, esquema de transmisión transparente.
		\item
			Permite las comunicaciones \vs{wdm} y \vs{dwdm} al ser acorde con su filosofía.
	\end{itemize}
\end{frame}

\begin{frame}{Esquema de comunicación transparente}
	\begin{itemize}[<+->]
		\item
			Toda la comunicación se realiza en el plano óptico.
		\item
			Transmisión entre fibras <<sin fisuras>> siguiendo el siguiente proceso:
		\begin{itemize}
			\item
				El haz se extrae directamente de la fibra.
			\item
				Se acopla al medio mediante unos colimadores controlados por mecanismos de haz.
			\item
				En recepción se realiza el proceso inverso y se inyecta la señal resultante en la fibra del otro extremo.
		\end{itemize}
	\end{itemize}
\end{frame}

\subsection*{Radio on Free Space Optics}

\begin{frame}{Motivos y funcionamiento}
	\begin{itemize}
		\item
			Sistemas desarrollados para dar cobertura allí donde las comunicación con \vs{rf} es difícil o está prohibida.
		\item
			Consiste en confinar las señales de \vs{rf} en portadoras ópticas y su transmisión mediante \vs{fso}.
		\item
			Funcionamiento similar a la tecnología \emph{Radio on Fiber} (RoF).
		\item
			Clasificación en sistemas indoor o outdoor.
	\end{itemize}
\end{frame}

\begin{frame}{Sistemas indoor}
	\begin{itemize}[<+->]
		\item
			Se pretende dar una cobertura similar a la ofrecida por los sistemas de \vs{rf}.
		\item
			Dos clasificaciones. Por tipo de enlace:
		\begin{itemize}
			\item
				De enlace directo.
			\item
				De enlace indirecto.
		\end{itemize}
		\item
			Según si es necesario un camino ininterrumpido entre emisor y receptor:
		\begin{itemize}
			\item
				Sistemas \vs{los}.
			\item
				Sistemas \vs{nlos}.
		\end{itemize}
	\end{itemize}
\end{frame}
