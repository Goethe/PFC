\subsection{Cálculos teóricos}

\begin{frame}{Matriz de Parámetros de Dispersión}
  La matriz de parámetros $S$ a la \alert{frecuencia de diseño} y cargando cada puerta con \alert{$Z_0$} es la siguiente:
  \begin{block}{Parámetros de dispersión, comportamiento deseado}
  \begin{equation}
	S=\frac{1}{\sqrt{2}}\paren{\begin{array}{rrrr}
	0 & -j & 0 & j\\
	-j & 0 &-j & 0\\
	0 & -j & 0 &-j\\
	j & 0 & -j & 0
	\end{array}}
  \end{equation}
  \end{block}
\end{frame}

\begin{frame}{Cálculo de $S_{11}$}
  \begin{itemize}
  	\item Se seleccionan las dos excitaciones siguientes:
  	\[
  	\mathrm{1\volado{a}\ excitacion}\ \left\{\begin{array}{c}
	  a^1_1=a\\
	  a^1_2=0\\
	  a^1_3=0\\
	  a^1_4=a
	\end{array}\right.\quad\mathrm{2\volado{a}\ excitacion}\ \left\{\begin{array}{c}
	  a^2_1=a\\
	  a^2_2=0\\
	  a^2_3=0\\
	  a^2_4=-a
	\end{array}\right.
  	\]
	\item Y este es el circuito resultante:
	\begin{figure}[ht]
		\begin{center}
		\includegraphics[keepaspectratio=true,width=110pt]{s11_1.1}
		\hspace{30pt}
		\includegraphics[keepaspectratio=true,width=110pt]{s11_2.1}
		\end{center}
	\caption{Cálculo de $S_{11}$\ \ \ a. 1ª excitación\ \ \ b. 2ª excitación}
	\end{figure}
  \end{itemize}
\end{frame}

\begin{frame}{Cálculo de $S_{12}$ y $S_{22}$}
  \begin{itemize}
  	\item Ahora se seleccionan las dos excitaciones siguientes
	\item Ambas idénticas para $S_{22}$ y $S_{12}$, al igual que el circuito
  	\[
  	\mathrm{1\volado{a}\ excitacion}\ \left\{\begin{array}{c}
	  a^1_1=0\\
	  a^1_2=a\\
	  a^1_3=a\\
	  a^1_4=0
	\end{array}\right.\quad\mathrm{2\volado{a}\ excitacion}\ \left\{\begin{array}{c}
	  a^2_1=0\\
	  a^2_2=a\\
	  a^2_3=-a\\
	  a^2_4=0
	\end{array}\right.
  	\]
	\item Y de nuevo este es el circuito resultante.
	\begin{figure}[ht]
		\begin{center}
		\includegraphics[keepaspectratio=true,width=110pt]{s22_1.1}
		\hspace{30pt}
		\includegraphics[keepaspectratio=true,width=110pt]{s22_2.1}
		\end{center}
	\caption{Cálculo de $S_{12}$\ \ \ a. 1ª excitación\ \ \ b. 2ª excitación}
	\end{figure}
  \end{itemize}
\end{frame}

\begin{frame}{Cálculo del resto de parámetros}
  \begin{itemize}
	\item Red \alert{simétrica} y \alert{recíproca}.
	\pause
	\item Una vez realizados los cálculos anteriores, el resto de parámetros se obtiene como combinación de los resultados obtenidos.
	\begin{block}{Parámetros $S$, primer grupo}
	\begin{equation}
		\begin{split}
		S_{11}&=\frac{1}{2}\paren{S^1_{11}+S^2_{11}}\qquad S_{22}=\frac{1}{2}\paren{S^1_{22}+S^2_{22}}\\
		S_{12}&=\frac{1}{2}\paren{S^1_{12}+S^2_{12}}\qquad S_{13}=\frac{1}{2}\paren{S^1_{12}-S^2_{12}}\\
		S_{23}&=\frac{1}{2}\paren{S^1_{22}-S^2_{22}}\qquad S_{14}=\frac{1}{2}\paren{S^1_{11}-S^2_{11}}
		\end{split}
	\end{equation}
	\end{block}
  \end{itemize}
\end{frame}

\begin{frame}{Cálculo del\ldots continuación}
	\begin{block}{Matriz calculada teóricamente}
	\begin{equation}
		\begin{split}
		S&=\paren{\begin{array}{cccc}
		S_{11} & S_{12} & S_{13} & S_{14}\\
		S_{21} & S_{22} & S_{23} & S_{24}\\
		S_{31} & S_{32} & S_{33} & S_{34}\\
		S_{41} & S_{42} & S_{43} & S_{44}
		\end{array}}\\
		\\
		&=\paren{\begin{array}{cccc}
		S_{11} & S_{12} & S_{13} & S_{14}\\
		S_{12} & S_{22} & S_{23} & S_{13}\\
		S_{13} & S_{23} & S_{22} & S_{12}\\
		S_{14} & S_{13} & S_{12} & S_{11}
		\end{array}}
		\end{split}
	\end{equation}
	\end{block}
\end{frame}

\subsection[Cálculo de $\epsilon_{\mathrm{rel}}$]{Cálculo de la permitividad relativa del dieléctrico}

\begin{frame}{Laboratorio, medidas}
  \begin{itemize}
	\item Para poder obtener un valor para \alert{$\epsilon_{\mathrm{rel}}$} tenemos en primer lugar que medir experimentalmente la \alert{$\epsilon_{\mathrm{eff}}$}.
	\item Para ese propósito empleamos una línea con un stub terminado en circuito abierto construida en tecnología microstrip, una línea como la mostrada en la \hyperref[fig:stub]{figura}.
	\begin{figure}[hb]\label{fig:stub}
		\begin{center}
		\includegraphics[keepaspectratio=true,width=125pt]{stub.1}
		\end{center}
	\caption{Línea microstrip con stub disponible en el laboratorio}
	\end{figure}
  \end{itemize}
\end{frame}

\begin{frame}{Cálculos -- Permitividad efectiva}
  \begin{itemize}
	\item Utilizamos la medida experimental de $f$ para obtener la $\epsilon_{\mathrm{eff}}$.
	\[
		\mathrm{Valores\ conocidos}\quad\left\{\begin{array}{lll}
		\lambda=17.2\ \mathrm{cm}\\
		c=3\cdot10^8\ \mathrm{m}\cdot\mathrm{s}^{-1}\\
		f=1.005\ \mathrm{GHz}
		\end{array}\right.
	\]
	\begin{block}{Permitividad efectiva}
	\begin{equation}
		\lambda=\frac{c}{f\sqrt{\epsilon_{\mathrm{eff}}}}\quad\rightarrow\quad\epsilon_{\mathrm{eff}}=\paren{\frac{c}{\lambda\cdot f}}^2=3.009\ \mathrm{F}/\mathrm{m}
	\end{equation}
	\end{block}
  \end{itemize}
\end{frame}

\begin{frame}{Cálculos -- Permitividad relativa}
  \begin{itemize}
	\item A partir de los siguientes datos del material y con la ayuda de \emph{AppCad} se obtiene el valor de la $\epsilon_{\mathrm{rel}}$.
	\[
		\mathrm{Valores\ conocidos}\quad\left\{\begin{array}{lll}
		H=800\ \mu\mathrm{m}\\
		W=500\ \mu\mathrm{m}\\
		T=30\ \mu\mathrm{m}
		\end{array}\right.
	\]
	\begin{block}{Permitividad relativa}
	\begin{equation}
		\epsilon_{\mathrm{rel}}=4.392\ \frac{\mathrm{F}}{\mathrm{m}}
	\end{equation}
	\end{block}
  \end{itemize}
\end{frame}
